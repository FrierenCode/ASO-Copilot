  How to verify

  Local dev setup

  # 1. Create the D1 database (one-time)
  cd apps/api
  pnpm wrangler d1 create aso-copilot
  # Paste the returned database_id into wrangler.jsonc

  # 2. Apply migration locally
  pnpm wrangler d1 migrations apply aso-copilot --local

  # 3. Create .dev.vars
  cat > apps/api/.dev.vars <<'EOF'
  UID_COOKIE_SECRET=a-random-32-char-string-here-ok!
  POLAR_WEBHOOK_SECRET=whsec_dGVzdA==
  EOF

  # 4. Run dev server
  pnpm dev

  Quota behaviour

  # First 3 calls succeed (free plan = 3/month)
  curl -s -c /tmp/jar.txt -b /tmp/jar.txt -X POST http://localhost:8787/generate \
    -H "Content-Type: application/json" \
    -d '{"appName":"X","category":"Productivity","screenshots":["a","b","c","d","e","f"]}' | jq   
  .status

  # 4th call returns 429
  # { "error": "LIMIT_EXCEEDED", "upgrade_url": "/pricing" }

  Entitlements response

  Webhook dedupe (simulated)

  # Send a webhook twice with the same webhook-id header
  BODY='{"type":"subscription.updated","data":{}}'
  curl -s -X POST http://localhost:8787/webhooks/polar \
    -H "webhook-id: test-wh-001" \
    -H "webhook-timestamp: $(date -u +%s)" \
    -H "webhook-signature: v1,INVALIDSIG" \
    -d "$BODY"
  # First call: stored as received → failed (bad sig)

  curl -s -X POST http://localhost:8787/webhooks/polar \
    -H "webhook-id: test-wh-001" \
    -H "webhook-timestamp: $(date -u +%s)" \
    -H "webhook-signature: v1,INVALIDSIG" \
    -d "$BODY"
  # Second call: { "ok": true, "deduped": true }  ← already processed/failed, skipped

  Run tests

  cd apps/api && pnpm test
  # 8/8 pass including quota enforcement, 409 dedup, and entitlements shape
